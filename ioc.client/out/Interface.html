Class: Interface

Interface(data)

new Interface(data)

Internet of Coins main API interface object
Parameters:
Name Type Description
data Object one of the three methods below must be passed.
Properties
Name Type Description
http Object a http object. (https://nodejs.org/api/http.html)
XMLHttpRequest Object a XMLHttpRequest object. (https://developer.mozilla.org/nl/docs/Web/API/XMLHttpRequest)
custom function a custom connector method which receives a hostname, query, dataCallback and errorCallback parameters and returns a string.

Examples
// Node JS
var IoC = require('./ioc.nodejs.client');

var ioc = new IoC.Interface({http:require('http')});

function onSucces(){
 console.error('Done.');
}

function onError(){
  console.error('Oops, something went wrong!');
}

ioc.init(null, onSucces, onError);
// Webpage
// add <script src="./ioc.web.client.js"></script> to html header

var ioc = new IoC.Interface({XMLHttpRequest:XMLHttpRequest});

function onSucces(){
 console.error('Done.');
}

function onError(){
  console.error('Oops, something went wrong!');
}

ioc.init(null, onSucces, onError);
var ioc = new IoC.Interface();

ioc.sequential([
  'init', // Initialize ioc
  {username: '****************', password: '****************'}, // Define credentials
  'login', // Do the login
  {host: 'http://localhost:1111/'}, // Define the host
  'addHost', // Add and initialize the host
  {symbol: 'dummy'}, // Define the asset
  'addAsset', // Add and initialize the asset
  {symbol: 'dummy', amount: 100}, // Define the transaction
  'transaction' // Execute the transaction
],
 onSucces,
 onError
);
